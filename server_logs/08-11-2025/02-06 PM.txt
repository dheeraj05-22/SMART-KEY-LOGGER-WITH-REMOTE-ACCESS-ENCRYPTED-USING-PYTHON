def view_logs_by_date(date):
    folder = log_dir / date
    if not folder.exists():
        return "Date folder not found", 404

    files = [f for f in folder.glob("*.txt") if f.is_file()]

    time_regex = re.compile(r'^(\d{2}-\d{2} [AP]M)')  # matches "01-48 PM" at start

    def parse_file_time(f: Path):
        name = f.name
        m = time_regex.match(name)
        if m:
            time_part = m.group(1)  # e.g. "01-48 PM"
            try:
                # combine with folder (date) to get a full datetime for correct ordering
                dt_str = f"{date} {time_part}"           # "08-11-2025 01-48 PM"
                return datetime.strptime(dt_str, "%d-%m-%Y %I-%M %p")
            except Exception:
                pass
        # fallback to file mtime if parse fails
        return datetime.fromtimestamp(f.stat().st_mtime)

    # newest first
    files.sort(key=parse_file_time, reverse=True)
